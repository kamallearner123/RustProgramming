Spin up an HTTP server with Actix Web, understand async, and build your first routes.

Concepts you'll master

  • Rust async runtime (tokio)
  • Actix handlers as async fn returning responder types
  • Routing basics and extracting params

Project Setup

rustup default stable
cargo new hello_web --bin
cd hello_web

Add Dependencies

[dependencies]
actix-web = "4"
tokio = { version = "1", features = ["rt-multi-thread", "macros"] }

Hello Server

use actix_web::{get, App, HttpServer, Responder};

#[get("/")]
async fn home() -> impl Responder { "Hello, Rust Web! ðŸ¦€" }

#[tokio::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| App::new().service(home))
        .bind(("127.0.0.1", 8080))?
        .run()
        .await
}

Run: 

cargo run
and open http://127.0.0.1:8080

Path & Query Params

use actix_web::{web, App, HttpServer, Responder};

async fn greet(path: web::Path<(String,)>) -> impl Responder {
    format!("Hi, {} ðŸ‘‹", path.into_inner().0)
}

async fn add(q: web::Query<std::collections::HashMap<String,String>>) -> impl Responder {
    let a: i32 = q.get("a").and_then(|s| s.parse().ok()).unwrap_or(0);
    let b: i32 = q.get("b").and_then(|s| s.parse().ok()).unwrap_or(0);
    format!("a+b = {}", a + b)
}

#[tokio::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| App::new()
        .route("/hi/{name}", web::get().to(greet))
        .route("/add", web::get().to(add)))
        .bind(("127.0.0.1",8080))?.run().await
}
